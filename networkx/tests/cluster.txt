>>> G = complete_graph(5)

Transitivity is weighted average of clustering
----------------------------------------------

>>> t1=transitivity(G)
>>> print t1
1.0
>>> (cluster_d2,weights)=clustering(G,with_labels=True,weights=True)
>>> trans=[]
>>> for v in G.nodes():
...   trans.append(cluster_d2[v]*weights[v])
>>> t2=sum(trans)
>>> abs(t1-t2)<1e-15 
True
